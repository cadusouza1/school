			             & = \frac{ 2^4 }{ 4 } - \frac{ 0^4 }{ 4 } & \\
			             & = 4                                     & \\
		            \end{flalign*}

		      \item \begin{flalign*}
			             & \int_{-1}^{1} (-1 - 2y - 3y^2) dy                                                                                                      & \\

			             & = -1 y \bigg\rvert_{-1}^1 - 2 \frac{ y^2 }{ 2 } \bigg\rvert_{-1}^1 - 3 \frac{ y^3 }{ 3 } \bigg\rvert_{-1}^1                            & \\
			             & = - [1 - (-1)] - 2 \left( \frac{ 1^2 }{ 2 } - \frac{ (-1)^2 }{ 2 } \right) - 3 \left( \frac{ 1^3 }{ 3 } - \frac{ (-1)^3 }{ 3 } \right) & \\
			             & = -2 - 0 + 2 = 0                                                                                                                       & \\
		            \end{flalign*}

		      \item \begin{flalign*}
			             & \int_{0}^{49} \sqrt{t} dt                                                    & \\
			             & = \int_{0}^{49} t^{\frac{ 1 }{ 2 }} dt                                       & \\
			             & = \frac{ t^{\frac{ 1 }{ 2 } + 1} }{ \frac{ 1 }{ 2 } + 1 } \bigg\rvert_0^{49} & \\
			             & = \frac{ t^{\frac{ 3 }{ 2 } } }{ \frac{ 3 }{ 2 } } \bigg\rvert_0^{49}        & \\
			             & = \frac{ 2 }{ 3 } t^{\frac{ 3 }{ 2 } } \bigg\rvert_0^{49}                    & \\
			             & = \frac{ 2 }{ 3 } \left( 49^{\frac{ 3 }{ 2 }} - 1 \right)                    & \\
			             & = \frac{ 2 }{ 3 } \left( 7^3 - 1 \right)                                     & \\
			             & = \frac{ 2 \times 342 }{ 3 }                                                 & \\
			             & = \frac{ 686 }{ 3 }                                                          & \\
		            \end{flalign*}

		      \item \begin{flalign*}
			             & \int_{1}^{3} \frac{ 2 }{ x^3 } dx                           & \\
			             & = 2 \int_{1}^{3} x^{-3} dx                                  & \\
			             & = 2 \frac{ x^{-2} }{ -2 } \bigg\rvert_1^3                   & \\
			             & = 2 \left( \frac{ 3^{-2} }{ -2 } - \frac{ 1 }{ -2 } \right) & \\
			             & = - \frac{ 1 }{ 9 } + 1                                     & \\
			             & = \frac{ 8 }{ 9 }                                           & \\
		            \end{flalign*}

		      \item \begin{flalign*}
			             & \int_{0}^{\frac{ \pi }{ 4 }} \sec^2(\theta) d\theta & \\
			             & = \tan(\theta) \bigg\rvert_0^{\frac{ \pi }{ 4 } }   & \\
			             & = \tan \left( \frac{ \pi }{ 4 } \right) - \tan(0)   & \\
			             & = 1 - 0 = 1                                         & \\
		            \end{flalign*}

		      \item \begin{flalign*}
			             & \int_{0}^{1} 2^x dx                                          & \\
			             & = \frac{ 2^x }{ \ln(2) } \bigg\rvert_0^1                     & \\
			             & = \left( \frac{ 2 }{ \ln(2) } - \frac{ 1 }{ \ln(2) } \right) & \\
			             & = \frac{ 1 }{ \ln(2) }                                       & \\
		            \end{flalign*}

		      \item \begin{flalign*}
			             & \int \left( z^2 + 1 + \frac{ 1 }{ z^2 + 1 } \right) dz    & \\
			             & = \int z^2 dz + \int 1 dz + \int \frac{ 1 }{ z^2 + 1 } dz & \\
			             & = \frac{ z^3 }{ 3 } + z + \arctan(z) + C                  & \\
		            \end{flalign*}

		      \item \begin{flalign*}
			             & \int (1 + \tan^2(s)) ds & \\

		            \end{flalign*}
