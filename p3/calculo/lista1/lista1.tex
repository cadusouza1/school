	\item \begin{enumerate}
		      \item \begin{flalign*}
			             & f(+x) = (+x)^2 + 1                  & \\
			             & f(-x) = (-x)^2 + 1 = x^2 + 1 = f(x) & \\
		            \end{flalign*}

		      \item
		            \begin{flalign*}
			             & f(+x) = |+x| = x        & \\
			             & f(-x) = |-x| = x = f(x) & \\
		            \end{flalign*}

		      \item
		            \begin{flalign*}
			             & f(+x) = (x - 1)^3 = x^3 - 3x^2 + 3x - 1                  & \\
			             & f(-x) = (-x - 1)^3 = (-x)^3 - 3(-x)^2 + 3(-x) - 1        & \\
			             & = -x^3 - 3x^2 - 3x - 1 = -(x^3 + 3x^2 + 3x + 1) \ne f(x) & \\
		            \end{flalign*}
	      \end{enumerate}
\end{enumerate}
