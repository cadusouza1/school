				                      \end{bmatrix} - \lambda I \right)                     & = 0               & \\
			            det\left( \begin{bmatrix}
					                      \frac{ 1 }{ 3 } - \lambda & 0 \\ 0 & \frac{ 1 }{ 3 } - \lambda
				                      \end{bmatrix} \right) & = 0               &               \\
			            \left( \frac{ 1 }{ 3 } - \lambda \right)^2                     & = 0               &  \\
			            \frac{ 1 }{ 3 } - \lambda                                      & = 0               &  \\
			            \lambda                                                        & = \frac{ 1 }{ 3 } &  \\
		            \end{flalign*}
		      \item \begin{flalign*}
			            det\left( \begin{bmatrix}
					                      2 & 0 \\ 0 & 5
				                      \end{bmatrix} - \lambda I \right)  & = 0                         & \\
			            det\left( \begin{bmatrix}
					                      2 - \lambda & 0 \\ 0 & 5 - \lambda
				                      \end{bmatrix} \right) & = 0                         &              \\
			            (2 - \lambda)(5 - \lambda)         & = 0                         &           \\
			            \lambda                            & = 2 \text{ ou } \lambda = 5 &           \\
		            \end{flalign*}
	      \end{enumerate}
